{{- /* Renders the page TOC from pre-built data in site.Data.toc (generated by scripts/extract-toc.mjs before the Hugo build). Each item is one of: {id, text, depth} — a link entry {label, depth} — a non-linked group label (bold) Falls back to Hugo's .TableOfContents for pages not in the data file. Called with the page as context (.). */ -}} {{- $items := index site.Data.toc .File.Path -}} {{- if not $items -}} {{- .TableOfContents -}} {{- else -}} {{- $html := "
    " -}} {{- $prevDepth := 0 -}} {{- $first := true -}} {{- range $items -}} {{- $d := .depth -}} {{- if $first -}} {{- $first = false -}} {{- else if gt $d $prevDepth -}} {{- range seq 1 (sub $d $prevDepth) -}} {{- $html = printf "%s
      " $html -}} {{- end -}} {{- else if lt $d $prevDepth -}} {{- range seq 1 (sub $prevDepth $d) -}} {{- $html = printf "%s
    " $html -}} {{- end -}} {{- $html = printf "%s" $html -}} {{- else -}} {{- $html = printf "%s" $html -}} {{- end -}} {{- if .label -}} {{- $html = printf "%s
  • %s" $html .label -}} {{- else -}} {{- $href := printf "%s#%s" $.RelPermalink .id -}} {{- $html = printf `%s
  • %s` $html $href .text -}} {{- end -}} {{- $prevDepth = $d -}} {{- end -}} {{- if not $first -}} {{- range seq 0 $prevDepth -}} {{- $html = printf "%s
" $html -}} {{- end -}} {{- end -}} {{ $html | safeHTML }} {{- end -}}
