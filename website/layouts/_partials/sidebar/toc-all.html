{{- /* Renders a combined TOC nav from all headings in the page — both regular page headings and headings inside tabs/steps shortcodes — in document order. Hugo generates .TableOfContents from the page AST before shortcodes execute, so shortcode headings cannot appear at the right position in .TableOfContents. This partial instead scans .Content (the fully rendered page HTML) for all h2-h6 elements in document order, interleaving tab group labels where the tab group actually sits in the content. Page heading text is extracted from the rendered HTML by stripping the Goldmark-generated anchor link and remaining tags. Tab heading text (including deduplicated IDs) comes from Page.Scratch "toc-extra" populated by the tabs shortcode. Expects to be called with the page as context (.). Renders
    ...
, or nothing when empty. */ -}} {{- $content := .Content -}} {{- $extras := .Scratch.Get "toc-extra" -}} {{- /* Build index: tab heading ID → {group, level, text} */ -}} {{- $tabIndex := dict -}} {{- $curGroup := "" -}} {{- range $extras -}} {{- if eq .type "group" -}} {{- $curGroup = .name -}} {{- else if eq .type "heading" -}} {{- $tabIndex = merge $tabIndex (dict .id (dict "group" $curGroup "level" .level "text" .text)) -}} {{- end -}} {{- end -}} {{- /* Walk all heading opening tags in .Content in document order */ -}} {{- $items := slice -}} {{- $lastGroup := "" -}} {{- range findRE `]*id="[^"]+"` $content -}} {{- $id := replaceRE `.*\bid="([^"]+)".*` "$1" . -}} {{- $level := replaceRE `]*id="%s"[^>]*>.*?` $id) $content -}} {{- $text := "" -}} {{- with index $fullTag 0 -}} {{- $text = replaceRE `]*>.*?` "" . | replaceRE `<[^>]+>` "" | strings.TrimSpace -}} {{- end -}} {{- if $text -}} {{- $items = $items | append (dict "type" "page-heading" "level" $level "id" $id "text" $text) -}} {{- end -}} {{- end -}} {{- end -}} {{- if $items -}} {{- $minLevel := 6 -}} {{- range $items -}} {{- if and (ne .type "group") (lt .level $minLevel) -}} {{- $minLevel = .level -}} {{- end -}} {{- end -}} {{- $md := "" -}} {{- range $items -}} {{- if eq .type "group" -}} {{- $md = printf "%s\n- **%s**" $md .name -}} {{- else if eq .type "page-heading" -}} {{- $count := sub .level $minLevel -}} {{- $indent := strings.Repeat $count " " -}} {{- $md = printf "%s\n%s- [%s](#%s)" $md $indent .text .id -}} {{- else if eq .type "tab-heading" -}} {{- $count := add 1 (sub .level $minLevel) -}} {{- $indent := strings.Repeat $count " " -}} {{- $md = printf "%s\n%s- [%s](#%s)" $md $indent .text .id -}} {{- end -}} {{- end -}} {{ $.RenderString $md }} {{- end -}}
