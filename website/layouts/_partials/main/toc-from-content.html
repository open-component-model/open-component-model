{{- /* Builds the "on this page" TOC from the page's rendered content instead of Hugo's .TableOfContents. Hugo renders shortcode output (even with percent notation) as separate markdown fragments, so headings inside the tabs and steps shortcodes never reach .TableOfContents or .Fragments — the rendered HTML is the only place where every heading exists in document order. Emits the same markup shape as .TableOfContents:
  • ...
      ...
Also fails the build when two headings on one page render the same id (e.g. the same heading text in two sibling tabs): the duplicate would make deep links and TOC entries ambiguous — rename one of the headings. Levels mirror markup.tableOfContents (startLevel 2, endLevel 3) in config/_default/markup.yaml. */ -}} {{- $startLevel := 2 -}} {{- $endLevel := 3 -}} {{- $headings := slice -}} {{- $seen := dict -}} {{- $pattern := `]*\sid="[^"]+"[^>]*>[^\n]*?` -}} {{- range findRE $pattern .Content -}} {{- $level := int (substr . 2 1) -}} {{- $id := replaceRE `^]*\sid="([^"]+)".*` "$1" . -}} {{- if index $seen $id -}} {{- errorf "%q: duplicate heading id %q — two headings on this page render the same anchor (often the same heading text in two tabs); rename one of them so deep links and TOC entries stay unique" $.Page.Path $id -}} {{- end -}} {{- $seen = merge $seen (dict $id true) -}} {{- if and (ge $level $startLevel) (le $level $endLevel) -}} {{- $text := replaceRE `^]*>([^\n]*?)$` "$1" . -}} {{- $headings = $headings | append (dict "level" $level "id" $id "text" $text) -}} {{- end -}} {{- end -}} {{- if $headings -}}
{{- $prev := sub $startLevel 1 -}} {{- range $h := $headings -}} {{- if gt $h.level $prev -}} {{- range seq (add $prev 1) $h.level }}
  • {{ end -}} {{- else if lt $h.level $prev -}} {{- range seq $h.level (sub $prev 1) }}
{{ end -}}
  • {{- else -}}
  • {{- end -}} {{ $h.text | safeHTML }} {{- $prev = $h.level -}} {{- end -}} {{- range seq $startLevel $prev }}
    • {{ end -}}
    {{- end -}}
